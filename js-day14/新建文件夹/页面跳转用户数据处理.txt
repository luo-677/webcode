var data = location.search.slice(1).split("&");
data.forEach(function(val) {
    tmp = val.split("=");
    switch (tmp[0]) {
        case "name":
            console.log(`用户名是${tmp[1]}`);
            break;
        case "passwd":
            console.log(`密码是${tmp[1]}`);
            break;
    }
})
