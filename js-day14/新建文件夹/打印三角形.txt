function double(n) {
    return n < 10 ? '0' + n : n;
}
var num = 1;
var arr = [];
for (var i = 0; i <= 4; i++) {
    arr[i] = [];
    for (var j = 0; j <= 4; j++) {
        arr[i][j] = num++;
        if (i >= j) {
            document.write(double(arr[i][j]) + '&nbsp;&nbsp;&nbsp;');
        }
    }
    document.write('<br>');
}
