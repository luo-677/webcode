<script>
    window.onload = function() {
        var link = document.getElementById('link');

        link.onclick = function(ev) {
            ev.preventDefault(); // DOM标准方法
            阻止默认行为有兼容问题
            ev = ev || event;

            // 兼容写法
            if (ev.preventDefault) {
                ev.preventDefault();
            } else {
                ev.returnValue = false; // ie
            }

            alert(1);
        }
    }
</script>
